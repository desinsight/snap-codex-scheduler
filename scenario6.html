// 음성 명령을 스케줄 작업으로 변환하는 코드
async function scheduleNightTask(voiceCommand) {
  // 1. 음성 명령에서 시간 및 작업 추출
  const { scheduledTime, taskType, targets } = await CommandParser.extract(voiceCommand);

  // 2. YAML 작업 정의 생성
  const yaml = YAMLGenerator.createSchedule({
    time: scheduledTime, // "23:00"
    tasks: [
      { type: 'backup', target: targets.drawing, storage: ['s3', 'supabase'] },
      { type: 'report', reportType: 've_comparison', source: targets.drawing }
    ]
  });

  // 3. MCP 스케줄러에 작업 등록
  const taskId = await MCPScheduler.register(yaml);

  // 4. FastAPI Task Queue에 작업 등록
  await TaskQueue.enqueue({
    id: taskId,
    executionTime: scheduledTime,
    priority: 'medium',
    retryPolicy: { maxAttempts: 3, interval: '5m' }
  });

  return { taskId, scheduledTime, confirmation: '자동화 작업 예약 완료' };
}
