// 클라이언트별 맞춤형 보고서 생성 및 자동 전송
async function generateAndSendClientReport(clientId, projectId, changes) {
  // 1. 클라이언트 설정 로드 (템플릿, 연락처, 접근권한)
  const clientSettings = await MCPClient.getSettings(clientId);

  // 2. 변경사항 처리 및 비용 계산
  const costComparison = await CostCalculator.compareChanges(changes);
  const veProposals = await VEGenerator.suggest(projectId, costComparison);

  // 3. 보고서 형식별 생성
  const reportData = {
    summary: await ReportSummarizer.generate(changes),
    costComparison,
    veProposals,
    client: clientSettings
  };

  const files = await Promise.all([
    ReportRenderer.toPDF(reportData),
    ReportRenderer.toExcel(costComparison)
  ]);

  // 4. 안전한 공유 링크 생성 및 전송
  const shareLink = await SecureShare.createLink({
    clientId,
    projectId,
    expiry: '14d',
    files
  });

  // 5. 알림 전송 및 로깅
  await Promise.all([
    Notifications.sendToClient(clientId, shareLink),
    Slack.notifyTeam(`클라이언트 ${clientSettings.name} 보고서 전송 완료`),
    Analytics.logReportGeneration(clientId, projectId)
  ]);

  return { shareLink, trackingId: shareLink.trackingId };
}
