// 대시보드 실시간 데이터 스트림 구현
import { create } from 'zustand';
import { createClient } from '@supabase/supabase-js';
import { MCPAnalytics } from '@mcp/analytics-sdk';

const useAgentStore = create(set => ({
  agents: [],
  timeline: [],
  suggestions: [],

  initializeRealtime: () => {
    const socket = new WebSocket('wss://api.snapcodex.com/agents');
    socket.onmessage = event => {
      const data = JSON.parse(event.data);
      set(state => ({
        agents: data.agents || state.agents,
        timeline: [data.event, ...state.timeline].slice(0, 20),
        suggestions: data.suggestions || state.suggestions
      }));
      MCPAnalytics.track('agent_status_update', data);
    };
  }
}));
