// 음성 명령 처리 및 보고서 생성 코드
async function processVoiceCommand(audioInput) {
  // 1. Whisper로 음성 명령을 텍스트로 변환
  const transcription = await WhisperProcessor.transcribe(audioInput);

  // 2. LangChain Agent로 의도 및 개체 추출
  const { intent, entities } = await LangChainAgent.analyze(transcription);

  // 3. 이전 견적과 현재 견적 비교
  const yesterday = await Database.getEstimate({ date: entities.date.yesterday });
  const today = await Database.getEstimate({ date: entities.date.today });
  const comparison = MCPCompare.analyze(yesterday, today);

  // 4. 클라이언트용 보고서 생성
  return ChartRenderer.createReport({
    data: comparison,
    type: 'client',
    output: ['pdf', 'dashboard']
  });
}
